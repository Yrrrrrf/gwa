-- =============================================================================
-- Xibalbá — Function Definitions (Tourism Domain)
-- Functions are read-only helpers. Mutations belong to events or queries.
-- Return shapes are client contracts — coordinate with Frontend before changing.
-- =============================================================================

-- ---------------------------------------------------------------------------
-- fn::t — i18n lookup with English fallback
-- Used everywhere a display string needs locale-aware resolution.
-- ---------------------------------------------------------------------------
DEFINE FUNCTION fn::t($key: string, $locale: string) {
    LET $trans = (SELECT VALUE value FROM locale_string WHERE key = $key AND locale = $locale LIMIT 1);
    IF array::len($trans) > 0 {
        RETURN $trans[0];
    } ELSE {
        LET $fallback = (SELECT VALUE value FROM locale_string WHERE key = $key AND locale = 'en' LIMIT 1);
        RETURN array::first($fallback) OR NONE;
    };
};

-- ---------------------------------------------------------------------------
-- fn::businesses_near — geo radius search
-- Primary data source for the Explorer map view.
-- Returns active+verified businesses within $radius_km, ordered by distance.
-- Each record is augmented with a computed distance_km field.
-- geo::distance returns meters; divide by 1000 for km.
-- ---------------------------------------------------------------------------
DEFINE FUNCTION fn::businesses_near($point: geometry<point>, $radius_km: decimal) {
    LET $radius_m = $radius_km * 1000.0dec;
    LET $businesses = (
        SELECT
            *,
            geo::distance(coordinates, $point) / 1000.0 AS distance_km
        FROM business
        WHERE is_active = true
          AND is_verified = true
          AND geo::distance(coordinates, $point) <= $radius_m
        ORDER BY distance_km ASC
    );
    RETURN $businesses;
};

-- ---------------------------------------------------------------------------
-- fn::search_businesses — BM25 full-text search with filters
-- $city_id and $category_id are optional filters (pass NONE to skip).
-- Results ordered by review_count DESC as a relevance proxy.
-- ---------------------------------------------------------------------------
DEFINE FUNCTION fn::search_businesses(
    $query:       string,
    $locale:      string,
    $city_id:     option<record<city>>,
    $category_id: option<record<category>>
) {
    LET $results = (
        SELECT * FROM business
        WHERE is_active = true
          AND is_verified = true
          AND (name @@ $query OR description @@ $query)
          AND ($city_id     = NONE OR city     = $city_id)
          AND ($category_id = NONE OR category = $category_id)
        ORDER BY review_count DESC
    );
    RETURN $results;
};

-- ---------------------------------------------------------------------------
-- fn::tourist_recommendations — collaborative filtering via graph traversal
-- 1. Find businesses this tourist has visited
-- 2. Find other tourists who visited those same businesses
-- 3. Find what else those similar tourists visited (excluding already-known)
-- 4. Rank by overlap count and return top $limit business IDs
-- This is the hackathon's strongest demo moment — pure graph, zero app logic.
-- ---------------------------------------------------------------------------
DEFINE FUNCTION fn::tourist_recommendations($user_id: record<user>, $limit: int) {
    -- Step 1: businesses this tourist has already visited
    LET $visited = SELECT VALUE ->visits->business FROM only $user_id;

    -- Step 2: all tourists who visited any of those businesses (flattened)
    LET $similar_raw = SELECT VALUE <-visits<-user FROM business WHERE id IN $visited;
    LET $similar_tourists = array::flatten($similar_raw);

    -- Step 3: what those tourists visited, excluding what our tourist already knows
    LET $recommendations = (
        SELECT out AS business, count() AS overlap
        FROM visits
        WHERE in IN $similar_tourists
          AND in  != $user_id
          AND out NOT IN $visited
        GROUP BY out
        ORDER BY overlap DESC
        LIMIT $limit
    );
    LET $biz_ids = $recommendations.business;
    RETURN SELECT * FROM business WHERE id IN $biz_ids;
};

-- ---------------------------------------------------------------------------
-- fn::popular_in_city — top businesses by engagement score
-- Score = review_count × rating. Feeds "trending" UI sections.
-- ---------------------------------------------------------------------------
DEFINE FUNCTION fn::popular_in_city($city_id: record<city>, $limit: int) {
    LET $businesses = (
        SELECT *, (review_count * (rating OR 0.0dec)) AS score
        FROM business
        WHERE city = $city_id
          AND is_active   = true
          AND is_verified = true
        ORDER BY score DESC
        LIMIT $limit
    );
    RETURN $businesses;
};

-- ---------------------------------------------------------------------------
-- fn::city_snapshot — point-in-time telemetry for a city
-- Used by the Vision dashboard and the fn::city_snapshot integration test.
-- ---------------------------------------------------------------------------
DEFINE FUNCTION fn::city_snapshot($city_id: record<city>) {
    LET $biz       = (SELECT id FROM business WHERE city = $city_id AND is_active = true AND is_verified = true);
    LET $biz_ids   = $biz.id;
    LET $week_ago  = time::now() - 7d;

    LET $total_reviews = (SELECT VALUE count() FROM review WHERE out IN $biz_ids)[0] OR 0;
    LET $visits_week   = (SELECT VALUE count() FROM visits WHERE out IN $biz_ids AND visited_at >= $week_ago)[0] OR 0;
    LET $top_cat_row   = (
        SELECT out.category AS cat, count() AS cnt
        FROM visits
        WHERE out IN $biz_ids
        GROUP BY out.category
        ORDER BY cnt DESC
        LIMIT 1
    )[0];

    RETURN {
        city:              $city_id,
        active_businesses: array::len($biz_ids),
        total_reviews:     $total_reviews,
        visits_this_week:  $visits_week,
        top_category:      $top_cat_row.cat OR NONE
    };
};

-- =============================================================================
-- Xibalbá — Event Definitions (Tourism Domain)
-- Events handle audit and notification side effects only.
-- Domain decisions stay in the application (Rust core/domain).
-- The activity table is the observable event log for all side effects.
-- =============================================================================

-- ---------------------------------------------------------------------------
-- on_review_created
-- When a tourist creates a review, notify the business owner.
-- Includes tourist country_code and locale so the owner knows who reviewed.
-- Also serves as the trigger point for the first-review milestone check.
-- ---------------------------------------------------------------------------
DEFINE EVENT on_review_created ON review WHEN $event = 'CREATE' THEN {
    LET $biz = $after.out;
    LET $owner = (SELECT VALUE in FROM manages WHERE out = $biz LIMIT 1)[0];

    -- Recalculate and update business stats (rating/review_count)
    -- Using GROUP ALL ensures we get a single result with the aggregates
    LET $stats = (SELECT 
        math::mean(rating) AS avg_rating,
        count()            AS total
        FROM review WHERE out = $biz GROUP ALL);
    
    UPDATE $biz SET 
        rating       = $stats[0].avg_rating OR 0,
        review_count = $stats[0].total OR 0;

    -- Notify owner of new review
    IF $owner != NONE {
        CREATE activity SET
            type            = 'review.received',
            target_owner    = $owner,
            target_business = $after.out,
            meta = {
                rating:           $after.rating,
                locale:           $after.locale,
                reviewer_country: $after.in.country_code
            },
            occurred_at = time::now();
    };

    -- First-review milestone: check if this is the very first review for this business
    LET $review_count = (SELECT VALUE count() FROM review WHERE out = $after.out)[0] OR 0;
    IF $review_count = 1 AND $owner != NONE {
        CREATE activity SET
            type            = 'review.first',
            target_owner    = $owner,
            target_business = $after.out,
            meta = { rating: $after.rating },
            occurred_at = time::now();
    };
};

-- ---------------------------------------------------------------------------
-- on_review_deleted
-- When a review is deleted, recalculate business stats.
-- ---------------------------------------------------------------------------
DEFINE EVENT on_review_deleted ON review WHEN $event = 'DELETE' THEN {
    LET $biz = $before.out;
    LET $stats = (SELECT 
        math::mean(rating) AS avg_rating,
        count()            AS total
        FROM review WHERE out = $biz GROUP ALL);
    
    UPDATE $biz SET 
        rating       = $stats[0].avg_rating OR 0,
        review_count = $stats[0].total OR 0;
};

-- ---------------------------------------------------------------------------
-- on_business_verified
-- When a business goes from unverified → verified, it appears on the tourist
-- map for the first time. The owner gets a milestone notification.
-- ---------------------------------------------------------------------------
DEFINE EVENT on_business_verified ON business
    WHEN $event = 'UPDATE' AND $after.is_verified = true AND ($before.is_verified = false OR $before.is_verified = NONE)
THEN {
    LET $owner = (SELECT VALUE in FROM manages WHERE out = $after.id LIMIT 1)[0];
    CREATE activity SET
        type            = 'business.verified',
        target_owner    = $owner,
        target_business = $after.id,
        meta = { name: $after.name },
        occurred_at = time::now();
};

-- ---------------------------------------------------------------------------
-- on_visit_recorded
-- When a tourist visits a business for the first time, give the owner a
-- first-visit milestone. Repeat visits are silently tracked with no event.
-- ---------------------------------------------------------------------------
DEFINE EVENT on_visit_recorded ON visits WHEN $event = 'CREATE' THEN {
    LET $prior = (SELECT VALUE count() FROM visits WHERE in = $after.in AND out = $after.out)[0] OR 0;
    IF $prior <= 1 {
        LET $owner = (SELECT VALUE in FROM manages WHERE out = $after.out LIMIT 1)[0];
        IF $owner != NONE {
            CREATE activity SET
                type            = 'visit.first',
                target_owner    = $owner,
                target_business = $after.out,
                meta = { source: $after.source },
                occurred_at = time::now();
        };
    };
};

#!/bin/sh
set -e

echo "Starting SurrealDB..."
surreal start \
    --log "${SURREAL_LOG:-info}" \
    --user "${SURREAL_USER:-root}" \
    --pass "${SURREAL_PASS}" \
    --bind "0.0.0.0:${SURREAL_PORT:-8000}" \
    "${SURREAL_PATH:-memory}" &
PID=$!

if [ "${SEED_ON_START:-true}" = "true" ]; then
    echo "Waiting for SurrealDB to accept connections..."
    sleep 3
    for i in 1 2 3 4 5; do
        if surreal sql \
            --endpoint "http://127.0.0.1:${SURREAL_PORT:-8000}" \
            -u "${SURREAL_USER:-root}" \
            -p "${SURREAL_PASS}" \
            --ns "${SURREAL_NS:-app}" \
            --db "${SURREAL_DB:-main}" \
            -q 'RETURN true;' > /dev/null 2>&1; then
            break
        fi
        sleep 2
    done

    echo "Running init-db.sh..."
    /scripts/init-db.sh
    echo "Initialization complete."
fi

wait $PID

#!/bin/sh
set -e

RED='\033[0;31m'
GREEN='\033[0;32m'
BLUE='\033[0;34m'
GRAY='\033[0;90m'
NC='\033[0m'
BOLD='\033[1m'
CHECK='✓'
CROSS='✗'
ARROW='→'
BULLET='•'

URL="${SURREAL_URL:-http://localhost:8000/sql}"
SURREAL_USER="${SURREAL_USER:-root}"
SURREAL_PASS="${SURREAL_PASS:-root}"
NS="${SURREAL_NS:-app}"
DB="${SURREAL_DB:-main}"

# ---------------------------------------------------------------------------
run_surql() {
    # REMOVED -f so we can actually see the error body
    curl -s -X POST "$URL" \
        -H "Accept: application/json" \
        -H "surreal-ns: $NS" \
        -H "surreal-db: $DB" \
        -u "$SURREAL_USER:$SURREAL_PASS" \
        --data-binary "@$1"
}

execute_surql_file() {
    local file="$1"
    local filename=$(basename "$file")
    local dirname=$(basename "$(dirname "$file")")
    printf "\t${GRAY}${BULLET} %s/%s${NC} " "$dirname" "$filename"

    # CAPTURE the output instead of discarding it
    RESPONSE=$(run_surql "$file")

    # SurrealDB returns errors in the JSON body even if HTTP is 200
    # So we check if the response contains "ERR"
    if echo "$RESPONSE" | grep -q '"status":"ERR"'; then
        printf "${RED}${CROSS}${NC}\n"
        printf "${RED}    %s${NC}\n" "$RESPONSE" # <--- THIS PRINTS THE RAW ERROR
        return 1
    else
        printf "${GREEN}${CHECK}${NC}\n"
    fi
}

# ---------------------------------------------------------------------------
main() {
    printf "${BOLD}Xibalbá — SurrealDB Initialization${NC}\n\n"

    # Provision NS and DB WITHOUT headers first
    printf "${BLUE}${ARROW} Provisioning Namespace/DB${NC}\n"
    RESPONSE=$(curl -s -X POST "$URL" \
        -H "Accept: application/json" \
        -u "$SURREAL_USER:$SURREAL_PASS" \
        -d "DEFINE NAMESPACE IF NOT EXISTS $NS; DEFINE DATABASE IF NOT EXISTS $DB ON NAMESPACE $NS;")
    
    if echo "$RESPONSE" | grep -q '"status":"ERR"'; then
        printf "\t${RED}${CROSS} Provisioning failed${NC}\n"
        printf "${RED}    %s${NC}\n" "$RESPONSE"
        exit 1
    fi
    printf "\t${GREEN}${CHECK} Namespace: %s, Database: %s${NC}\n\n" "$NS" "$DB"

    local base="/init"

    for dir in $(find "$base" -mindepth 1 -maxdepth 1 -type d | sort); do
        printf "${BLUE}${ARROW} %s${NC}\n" "$(basename "$dir")"
        for f in $(find "$dir" -name "*.surql" | sort); do
            # The logic stays the same: if it returns 1, the script exits
            execute_surql_file "$f" || exit 1
        done
        printf "\n"
    done

    printf "${GREEN}${CHECK} Initialization complete — %s/%s${NC}\n\n" "$NS" "$DB"
}

main
